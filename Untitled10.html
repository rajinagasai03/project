import pandas as pd

# Read the CSV data
data = pd.read_csv('f:\jio\employee_burnout_analysis.csv')

# Count the occurrences of each gender
gender_counts = data['Gender'].value_counts()

# Create a bar graph

plt.title('Employee Gender Distribution')
